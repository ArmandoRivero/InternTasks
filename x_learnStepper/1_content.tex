\section{Objectives}
%******************************************************************************
Learn about stepper motors

\begin{enumerate}
  \item \textbf{What is a stepper motor?}  (half day)
  	\begin{todolist}
		\item Description of the kinds of steppers (Unipolar - Bipolar).
 		\item Elaborate a small research of how stepper motors work.
	\end{todolist}
  \item \textbf{How to use a stepper motor?} (half day)
  	\begin{todolist}
		\item How to control each kind of motors? 
		\item Which are the control sequences for each stepper?
		\item Identify the HW needed to connect the Stepper motor to an Arduino. (hint: LM293/LM293D)
	\end{todolist}

  \item \textbf{Programming the sequence using Arduino:} (one day)
  	\begin{todolist}
  		\item Using PINx of the Arduino's API
  		\begin{todolist}
  			\item Stepper in clockwise way direction
  			\item Stepper in counter clockwise way direction
  			\item Stepper in both directions
  		\end{todolist}
  		\item Using the PORTx of the microcontroller	
  		\begin{todolist}
  			\item Stepper in clockwise way direction
  			\item Stepper in counter clockwise way direction
  			\item Stepper in both directions
  		\end{todolist}
  	\end{todolist}

  \item \textbf{Optimise the program Part 1} (one day)
  	\begin{todolist}
  		\item Understand the use of push buttons using Arduino.
  		\item Identify the bouncing effect.
  		\item Solve the problem using "anti-bouncing" techniques.
  		\item Add the use of push buttons to define the direction of the stepper motor (left and right).
  		\item Add the use of push buttons to set the velocity of the stepper motor.  
  	\end{todolist}
  \item \textbf{Optimise the program Part 2} (one day)
  	\begin{todolist}
  		\item Introduce sensors to identify the position of the motor.
  	
  	\end{todolist}

  \item \textbf{Write the report using the \LaTeX \ template provided.}
\end{enumerate}
